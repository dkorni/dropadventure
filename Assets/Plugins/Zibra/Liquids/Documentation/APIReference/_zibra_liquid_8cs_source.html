1461 public void InitializeSimulation()
1462 {
1463 if (Initialized)
1464 {
1465 return;
1466 }
1467
1468 try
1469 {
1470#if UNITY_IOS
1471 if (SystemInfo.graphicsDeviceName == "Apple iOS simulator GPU")
1472 {
1473 throw new Exception("Zibra Liquids doesn't support iOS simulator. " +
1474 "Liquid was disabled.");
1475 }
1476#endif
1477
1478#if ZIBRA_LIQUID_PRO_VERSION && !ZIBRA_LIQUID_PRO_VERSION_NO_LICENSE_CHECK && UNITY_EDITOR
1479 if (!ZibraServerAuthenticationManager.GetInstance().IsLicenseVerified())
1480 {
1481 string errorMessage = "License wasn't verified. " +
1482 ZibraServerAuthenticationManager.GetInstance().GetErrorMessage() +
1483 " Liquid won't run in editor.";
1484 throw new Exception(errorMessage);
1485 }
1486#endif
1487
1488 // Copy Rendering Mode to internal variable to prevent user from modifying it during liquid lifetime
1489 ActiveRenderingMode = CurrentRenderingMode;
1490
1491#if UNITY_PIPELINE_HDRP
1492 if (RenderPipelineDetector.GetRenderPipelineType() == RenderPipelineDetector.RenderPipeline.HDRP)
1493 {
1494 bool missingRequiredParameter = false;
1495
1496 if (CustomLightHDRP == null && ActiveRenderingMode != RenderingMode.UnityRender)
1497 {
1498 Debug.LogError("No Custom Light set in Zibra Liquid.");
1499 missingRequiredParameter = true;
1500 }
1501
1502 if (ReflectionProbeHDRP == null && ActiveRenderingMode != RenderingMode.UnityRender)
1503 {
1504 Debug.LogError("No reflection probe added to Zibra Liquid.");
1505 missingRequiredParameter = true;
1506 }
1507
1508 if (missingRequiredParameter)
1509 {
1510 throw new Exception("Liquid creation failed due to missing parameter.");
1511 }
1512 }
1513#endif
1514
1515#if ZIBRA_LIQUID_PAID_VERSION
1516 if (InitialState == ZibraLiquid.InitialStateType.NoParticles || BakedInitialStateAsset == null)
1517#endif
1518 {
1519 bool haveEmitter = false;
1520 foreach (var manipulator in Manipulators)
1521 {
1522 if (manipulator.GetManipulatorType() == Manipulator.ManipulatorType.Emitter)
1523 {
1524 haveEmitter = true;
1525 break;
1526 }
1527 }
1528
1529 if (!haveEmitter)
1530 {
1531#if ZIBRA_LIQUID_PAID_VERSION
1532 throw new Exception("Liquid creation failed. Liquid have neither initial state nor emitters.");
1533#else
1534 throw new Exception("Liquid creation failed. Liquid have don't have any emitters.");
1535#endif
1536 }
1537 }
1538
1539 Camera.onPreRender += RenderCallBackWrapper;
1540
1541 SolverCommandBuffer = new CommandBuffer { name = "ZibraLiquid.Solver" };
1542
1543 CurrentInstanceID = NextInstanceId++;
1544
1545 ForceCloseCommandEncoder(SolverCommandBuffer);
1546 ZibraLiquidBridge.SubmitInstanceEvent(SolverCommandBuffer, CurrentInstanceID,
1547 ZibraLiquidBridge.EventID.CreateFluidInstance);
1548 Graphics.ExecuteCommandBuffer(SolverCommandBuffer);
1549 SolverCommandBuffer.Clear();
1550
1551 InitializeParticles();
1552
1553 var initializeGPUReadbackParamsBridgeParams = new InitializeGPUReadbackParams();
1554#if ZIBRA_LIQUID_FREE_VERSION
1555 UInt32 manipSize = 0;
1556#else
1557 UInt32 manipSize = (UInt32)ManipulatorManager.Elements * STATISTICS_PER_MANIPULATOR * sizeof(Int32);
1558#endif
1559 initializeGPUReadbackParamsBridgeParams.readbackBufferSize = sizeof(Int32) + manipSize;
1560 switch (SystemInfo.graphicsDeviceType)
1561 {
1562 case GraphicsDeviceType.Direct3D11:
1563 case GraphicsDeviceType.XboxOne:
1564 case GraphicsDeviceType.Switch:
1565#if UNITY_2020_3_OR_NEWER
1566 case GraphicsDeviceType.Direct3D12:
1567 case GraphicsDeviceType.XboxOneD3D12:
1568#endif
1569 initializeGPUReadbackParamsBridgeParams.maxFramesInFlight = QualitySettings.maxQueuedFrames + 1;
1570 break;
1571 default:
1572 initializeGPUReadbackParamsBridgeParams.maxFramesInFlight = (int)MaxFramesInFlight;
1573 break;
1574 }
1575
1576 IntPtr nativeCreateInstanceBridgeParams =
1577 Marshal.AllocHGlobal(Marshal.SizeOf(initializeGPUReadbackParamsBridgeParams));
1578 Marshal.StructureToPtr(initializeGPUReadbackParamsBridgeParams, nativeCreateInstanceBridgeParams, true);
1579
1580 SolverCommandBuffer.Clear();
1581 ZibraLiquidBridge.SubmitInstanceEvent(SolverCommandBuffer, CurrentInstanceID,
1582 ZibraLiquidBridge.EventID.InitializeGpuReadback,
1583 nativeCreateInstanceBridgeParams);
1584 Graphics.ExecuteCommandBuffer(SolverCommandBuffer);
1585
1586 ToFreeOnExit.Add(nativeCreateInstanceBridgeParams);
1587
1588 InitializeSolver();
1589
1590 Initialized = true;
1591
1592#if UNITY_EDITOR
1593 ZibraLiquidAnalyticsData.TrackLiquidInitialization(this);
1594#endif
1595 }
1596 catch (Exception)
1597 {
1598 ClearRendering();
1599 ClearSolver();
1600 throw;
1601 }
1602 }
