428 internal void ValidateParameters()
429 {
430#if ZIBRA_LIQUID_PRO_VERSION
431 if (Application.isPlaying)
432 {
433 if (MaximumSpeciesCount == 0)
434 {
435 MaximumSpeciesCount = AdditionalParticleSpecies.Count + MAX_RUNTIME_ADDED_SPECIES;
436 }
437 else
438 {
439 if (AdditionalParticleSpecies.Count >= MaximumSpeciesCount)
440 {
441 int toDelete = AdditionalParticleSpecies.Count - MaximumSpeciesCount + 1;
442 AdditionalParticleSpecies.RemoveRange(MaximumSpeciesCount - 1, toDelete);
443 Debug.LogWarning("Zibra Liquids: Exceeded maximum number of species in current instance");
444 }
445 }
446 }
447
448 foreach (var param in AdditionalParticleSpecies)
449 {
450 int ZeroParameterCount = 0;
451 if (param.Gravity == Vector3.zero)
452 ZeroParameterCount++;
453 ValidateGravity(ref param.Gravity);
454 if (param.FluidStiffness == 0.0f)
455 {
456 ZeroParameterCount++;
457 param.FluidStiffness = DEFAULT_STIFFNESS;
458 }
459 if (param.ParticleDensity == 0.0f)
460 {
461 ZeroParameterCount++;
462 param.ParticleDensity = DEFAULT_DENSITY;
463 }
464 if (param.Viscosity == 0.0f)
465 ZeroParameterCount++;
466 if (param.MaximumVelocity == 0.0f)
467 ZeroParameterCount++;
468
469 if (ZeroParameterCount == 5) // if parameters are initialized as 0
470 {
471 param.MaximumVelocity = DEFAULT_MAX_VELOCITY;
472 param.Viscosity = DEFAULT_VISCOSITY;
473 param.Gravity = new Vector3(0.0f, DEFAULT_GRAVITY, 0.0f);
474 }
475 }
476#endif
477
478 ValidateGravity(ref Gravity);
479 }
480
481 private void ValidateGravity(ref Vector3 gravity)
482 {
483 gravity.x = Mathf.Clamp(gravity.x, -GRAVITY_THRESHOLD, GRAVITY_THRESHOLD);
484 gravity.y = Mathf.Clamp(gravity.y, -GRAVITY_THRESHOLD, GRAVITY_THRESHOLD);
485 gravity.z = Mathf.Clamp(gravity.z, -GRAVITY_THRESHOLD, GRAVITY_THRESHOLD);
486 }
