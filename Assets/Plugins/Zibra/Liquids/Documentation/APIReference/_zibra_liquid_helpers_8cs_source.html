30 static string[] GetSupportedFormats()
31 {
32 List<string> result = new List<string>();
33 GraphicsFormat[] formatsToTest = new GraphicsFormat[]
34 {
35 GraphicsFormat.R16_SFloat,
36 GraphicsFormat.R32_SFloat,
37 GraphicsFormat.R8G8B8A8_UNorm,
38 GraphicsFormat.R16G16B16A16_SFloat,
39 GraphicsFormat.R32G32B32A32_SFloat,
40 GraphicsFormat.B10G11R11_UFloatPack32,
41 };
42
43 FormatUsage[] usagesToTest = new FormatUsage[]
44 {
45 FormatUsage.Sample,
46 FormatUsage.Render,
47 FormatUsage.Blend,
48 FormatUsage.ReadPixels,
49 FormatUsage.LoadStore,
50 };
51
52 foreach (GraphicsFormat format in formatsToTest)
53 {
54 string usages = "";
55 foreach (FormatUsage usage in usagesToTest)
56 {
57 if (SystemInfo.IsFormatSupported(format, usage))
58 {
59 usages += $".{usage}";
60 }
61 }
62 if (usages == ".Sample.Render.Blend.ReadPixels.LoadStore")
63 {
64 usages = ".All";
65 }
66 if (usages == "")
67 {
68 usages = ".Unsupported";
69 }
70 result.Add($"{format}{usages}");
71 }
72
73 return result.ToArray();
74 }
